```dart
int? _parseTimeCode(String input) {
  input = input.trim();
  if (input.isEmpty) return null;

  // 分离毫秒部分
  String mainPart = input;
  int millis = 0;
  if (input.contains('.')) {
    final dotParts = input.split('.');
    mainPart = dotParts[0];
    final msStr = dotParts[1];
    if (msStr.length == 1) {
      millis = int.parse(msStr) * 100;
    } else if (msStr.length == 2) {
      millis = int.parse(msStr) * 10;
    } else {
      millis = int.tryParse(msStr.padRight(3, '0').substring(0, 3)) ?? 0;
    }
  }

  final parts = mainPart.split(':');
  try {
    if (parts.length == 1) {
      final seconds = int.parse(parts[0]);
      return seconds * 1000 + millis;
    } else if (parts.length == 2) {
      final minutes = int.parse(parts[0]);
      final seconds = int.parse(parts[1]);
      return (minutes * 60 + seconds) * 1000 + millis;
    } else if (parts.length == 3) {
      final hours = int.parse(parts[0]);
      final minutes = int.parse(parts[1]);
      final seconds = int.parse(parts[2]);
      return (hours * 3600 + minutes * 60 + seconds) * 1000 + millis;
    }
  } catch (_) {
    return null;
  }
  return null;
}
```

```dart
  final frameMs = (1000 / video.frameRate).round();
  _previewKey.currentState?.seekTo((pos - frameMs).clamp(0, video.duration));
}
```

```dart
  final videoId = await ref.read(queueProvider.notifier).resetTask(taskId);
  if (videoId != null && mounted) {
```
